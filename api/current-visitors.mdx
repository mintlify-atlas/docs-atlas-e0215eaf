## Overview

The `current_visitors` endpoint returns the number of unique visitors (sessions) in the last 5 minutes. This is ideal for displaying realtime activity on your dashboard.

## Endpoint

```
GET https://api.tinybird.co/v0/pipes/current_visitors.json
```

## Parameters

<ParamField query="tenant_id" type="string" optional>
  Filter by tenant ID
</ParamField>

<ParamField query="domain" type="string" optional>
  Filter by domain
</ParamField>

## SQL Query

The endpoint executes the following SQL query:

```sql
SELECT uniq(session_id) AS visits
FROM analytics_hits
WHERE timestamp >= (now() - interval 5 minute)
    {% if defined(tenant_id) %}
    AND tenant_id = {{ String(tenant_id, description="Filter by tenant ID") }}
    {% end %}
    {% if defined(domain) %}
    AND domain = {{ String(domain, description="Filter by domain") }}
    {% end %}
```

## Response

<ResponseField name="visits" type="uint64" required>
  Number of unique visitors in the last 5 minutes
</ResponseField>

## TypeScript Usage

```typescript
import { Tinybird } from '@tinybirdco/sdk';
import type { CurrentVisitorsParams, CurrentVisitorsOutput } from './endpoints';

const tb = new Tinybird({ token: process.env.TINYBIRD_TOKEN });

// Get current visitors for all domains
const result = await tb.exec<CurrentVisitorsOutput>('current_visitors');
console.log(`Current visitors: ${result.data[0].visits}`);

// Get current visitors for a specific domain
const params: CurrentVisitorsParams = {
  domain: 'example.com'
};
const domainResult = await tb.exec<CurrentVisitorsOutput>('current_visitors', params);
console.log(`Current visitors for ${params.domain}: ${domainResult.data[0].visits}`);

// Get current visitors for a specific tenant
const tenantParams: CurrentVisitorsParams = {
  tenant_id: 'tenant-123',
  domain: 'example.com'
};
const tenantResult = await tb.exec<CurrentVisitorsOutput>('current_visitors', tenantParams);
```
